 57     /*
 58     *  attachment_fields_to_edit
 59     *
 60     *  Adds ACF fields to the attachment form fields
 61     *
 62     *  @type    filter
 63     *  @date    14/07/13
 64     *
 65     *  @param   {array}     $form_fields
 66     *  @return  {object}    $post
 67     */
 68     
 69     function attachment_fields_to_edit( $form_fields, $post ) 
 70     {
 71         // vars
 72         $screen = get_current_screen();
 73         $post_id = $post->ID;
 74         
 75         
 76         if( !empty($screen) )
 77         {
 78             return $form_fields;
 79         }
 80         
 81         
 82         // get field groups
 83         $filter = array( 'post_type' => 'attachment' );
 84         $metabox_ids = array();
 85         $metabox_ids = apply_filters( 'acf/location/match_field_groups', $metabox_ids, $filter );
 86         
 87         
 88         // validate
 89         if( empty($metabox_ids) )
 90         {
 91             return $form_fields;    
 92         }
 93         
 94         
 95         $acfs = apply_filters('acf/get_field_groups', array());
 96     
 97     
 98         if( is_array($acfs) ){ foreach( $acfs as $acf ){
 99             
100             // only add the chosen field groups
101             if( !in_array( $acf['id'], $metabox_ids ) )
102             {
103                 continue;
104             }
105             
106             
107             // load fields
108             $fields = apply_filters('acf/field_group/get_fields', array(), $acf['id']);
109             
110             
111             if( is_array($fields) ){ foreach( $fields as $i => $field ){
112                 
113                 // if they didn't select a type, skip this field
114                 if( !$field || !$field['type'] || $field['type'] == 'null' )
115                 {
116                     continue;
117                 }
118                     
119             
120                 // set value
121                 if( !isset($field['value']) )
122                 {
123                     $field['value'] = apply_filters('acf/load_value', false, $post_id, $field);
124                     $field['value'] = apply_filters('acf/format_value', $field['value'], $post_id, $field);
125                 }
126                 
127                 
128                 // create field
129                 $field['name'] = 'fields[' . $field['key'] . ']';
130                 
131                 ob_start();
132                     
133                     do_action('acf/create_field', $field);
134                     
135                     $html = ob_get_contents();
136                 
137                 ob_end_clean();
138  
139                 
140                 $form_fields[ $field['name'] ] = array(
141                     'label' => $field['label'],
142                     'input' => 'html',
143                     'html' => $html
144                 );
145                 
146             }};
147  
148             
149         }}
150         
151         
152         // return
153         return $form_fields;
154     }

643         
644         // metabox ids is a string with commas
645         $options['metabox_ids'] = explode( ',', $options['metabox_ids'] );
646         
647             
648         // get acfs
649         $acfs = apply_filters('acf/get_field_groups', false);
650             
651         
652         // layout

659         {
660             $layout = 'metabox';
661         }
662         
663         
664         if( $acfs )
665         {
666             foreach( $acfs as $acf )
667             {
668                 // load options
669                 $acf['options'] = apply_filters('acf/field_group/get_options', array(), $acf['id']);
670                 
671                 
672                 // only add the chosen field groups
673                 if( !in_array( $acf['id'], $options['metabox_ids'] ) )
674                 {

677                 
678                 
679                 // layout dictates heading
680                 $title = true;
681

685                 }
686                 
687

692                     {
693                         echo '<h3>' .$acf['title'] . '</h3>';
694                     }
695                     
696                     echo '<table class="form-table"><tbody>';
697                 }
698                 
699                 
700                 // wrapper
701                 if( $layout == 'tr' )
702                 {

755                                 echo $field['instructions'];
756                             echo '</p>';

758                             $field['name'] = 'fields[' . $field['key'] . ']';
