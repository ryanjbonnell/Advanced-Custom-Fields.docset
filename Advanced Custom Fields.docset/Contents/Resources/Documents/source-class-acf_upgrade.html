 41     function admin_menu()
 42     {
 43         // dont run on plugin activate!
 44         if( isset($_GET['action']) && $_GET['action'] == 'activate-plugin' )
 45         {
 46             return;
 47         }
 48 
 49 
 50         // vars
 51         $new_version = apply_filters('acf/get_info', 'version');
 52         $old_version = get_option('acf_version', false);
 53 
 54 
 55         if( $new_version != $old_version )
 56         {
 57             update_option('acf_version', $new_version );
 58 
 59             if( !$old_version )
 60             {
 61                 // do nothing, this is a fresh install
 62             }
 63             elseif( $old_version < '4.0.0' && $new_version >= '4.0.0')
 64             {
 65                 $url = admin_url('edit.php?post_type=acf&info=whats-new');
 66                 wp_redirect( $url );
 67                 exit;
 68 
 69             }
 70         }
 71         
 72         
 73         // update admin page
 74         add_submenu_page('edit.php?post_type=acf', __('Upgrade','acf'), __('Upgrade','acf'), 'manage_options','acf-upgrade', array($this,'html') );
 75     }
