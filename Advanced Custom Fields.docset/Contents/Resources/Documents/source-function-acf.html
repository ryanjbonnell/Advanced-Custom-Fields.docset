102     /*
103     *  helpers_get_dir
104     *
105     *  This function will calculate the directory (URL) to a file
106     *
107     *  @type    function
108     *  @date    30/01/13
109     *  @since   3.6.0
110     *
111     *  @param   $file (file) a reference to the file
112     *  @return  (string)
113     */
114     
115     function helpers_get_dir( $file )
116     {
117         $dir = trailingslashit(dirname($file));
118         $count = 0;
119         
120         
121         // sanitize for Win32 installs
122         $dir = str_replace('\\' ,'/', $dir); 
123         
124         
125         // if file is in plugins folder
126         $wp_plugin_dir = str_replace('\\' ,'/', WP_PLUGIN_DIR); 
127         $dir = str_replace($wp_plugin_dir, plugins_url(), $dir, $count);
128         
129         
130         if( $count < 1 )
131         {
132             // if file is in wp-content folder
133             $wp_content_dir = str_replace('\\' ,'/', WP_CONTENT_DIR); 
134             $dir = str_replace($wp_content_dir, content_url(), $dir, $count);
135         }
136         
137         
138         if( $count < 1 )
139         {
140             // if file is in ??? folder
141             $wp_dir = str_replace('\\' ,'/', ABSPATH); 
142             $dir = str_replace($wp_dir, site_url('/'), $dir);
143         }
144         
145 
146         return $dir;
147     }

150     /*
151     *  acf/get_post_id
152     *
153     *  A helper function to filter the post_id variable.
154     *
155     *  @type    filter
156     *  @date    27/05/13
157     *
158     *  @param   {mixed} $post_id
159     *  @return  {mixed} $post_id
160     */
161     
162     function get_post_id( $post_id )
163     {
164         // set post_id to global
165         if( !$post_id )
166         {
167             global $post;
168             
169             if( $post )
170             {
171                 $post_id = intval( $post->ID );
172             }
173         }
174         
175         
176         // allow for option == options
177         if( $post_id == "option" )
178         {
179             $post_id = "options";
180         }
181         
182         
183         // object
184         if( is_object($post_id) )
185         {
186             if( isset($post_id->roles, $post_id->ID) )
187             {
188                 $post_id = 'user_' . $post_id->ID;
189             }
190             elseif( isset($post_id->taxonomy, $post_id->term_id) )
191             {
192                 $post_id = $post_id->taxonomy . '_' . $post_id->term_id;
193             }
194             elseif( isset($post_id->ID) )
195             {
196                 $post_id = $post_id->ID;
197             }
198         }
199         
200         
201         /*
202         *  Override for preview
203         *  
204         *  If the $_GET['preview_id'] is set, then the user wants to see the preview data.
205         *  There is also the case of previewing a page with post_id = 1, but using get_field
206         *  to load data from another post_id.
207         *  In this case, we need to make sure that the autosave revision is actually related
208         *  to the $post_id variable. If they match, then the autosave data will be used, otherwise, 
209         *  the user wants to load data from a completely different post_id
210         */
211         
212         if( isset($_GET['preview_id']) )
213         {
214             $autosave = wp_get_post_autosave( $_GET['preview_id'] );
215             if( $autosave->post_parent == $post_id )
216             {
217                 $post_id = intval( $autosave->ID );
218             }
219         }
220         
221         
222         // return
223         return $post_id;
224     }

265     /*
266     *  parse_types
267     *
268     *  @description: helper function to set the 'types' of variables
269     *  @since: 2.0.4
270     *  @created: 9/12/12
271     */
272     
273     function parse_types( $value )
274     {
275         // vars
276         $restricted = array(
277             'label',
278             'name',
279             '_name',
280             'value',
281             'instructions'
282         );
283         
284         
285         // is value another array?
286         if( is_array($value) )
287         {
288             foreach( $value as $k => $v )
289             {
290                 // bail early for restricted pieces
291                 if( in_array($k, $restricted, true) )
292                 {
293                     continue;
294                 }
295                 
296                 
297                 // filter piece
298                 $value[ $k ] = apply_filters( 'acf/parse_types', $v );
299             }   
300         }
301         else
302         {
303             // string
304             if( is_string($value) )
305             {
306                 $value = trim( $value );
307             }
308             
309             
310             // numbers
311             if( is_numeric($value) )
312             {
313                 // check for non numeric characters
314                 if( preg_match('/[^0-9]/', $value) )
315                 {
316                     // leave value if it contains such characters: . + - e
317                     //$value = floatval( $value );
318                 }
319                 else
320                 {
321                     $value = intval( $value );
322                 }
323             }
324         }
325         
326         
327         // return
328         return $value;
329     }

636     /*
637     *  get_taxonomies_for_select
638     *
639     *  @description: 
640     *  @since: 3.6
641     *  @created: 27/01/13
642     */
643     
644     function get_taxonomies_for_select( $choices, $simple_value = false )
645     {   
646         // vars
647         $post_types = get_post_types();
648         
649         
650         if($post_types)
651         {
652             foreach($post_types as $post_type)
653             {
654                 $post_type_object = get_post_type_object($post_type);
655                 $taxonomies = get_object_taxonomies($post_type);
656                 if($taxonomies)
657                 {
658                     foreach($taxonomies as $taxonomy)
659                     {
660                         if(!is_taxonomy_hierarchical($taxonomy)) continue;
661                         $terms = get_terms($taxonomy, array('hide_empty' => false));
662                         if($terms)
663                         {
664                             foreach($terms as $term)
665                             {
666                                 $value = $taxonomy . ':' . $term->term_id;
667                                 
668                                 if( $simple_value )
669                                 {
670                                     $value = $term->term_id;
671                                 }
672                                 
673                                 $choices[$post_type_object->label . ': ' . $taxonomy][$value] = $term->name; 
674                             }
675                         }
676                     }
677                 }
678             }
679         }
680         
681         return $choices;
682     }

685     /*
686     *  get_post_types
687     *
688     *  @description: 
689     *  @since: 3.5.5
690     *  @created: 16/12/12
691     */
692     
693     function get_post_types( $post_types, $exclude = array(), $include = array() )
694     {
695         // get all custom post types
696         $post_types = array_merge($post_types, get_post_types());
697         
698         
699         // core include / exclude
700         $acf_includes = array_merge( array(), $include );
701         $acf_excludes = array_merge( array( 'acf', 'revision', 'nav_menu_item' ), $exclude );
702      
703         
704         // include
705         foreach( $acf_includes as $p )
706         {                   
707             if( post_type_exists($p) )
708             {                           
709                 $post_types[ $p ] = $p;
710             }
711         }
712         
713         
714         // exclude
715         foreach( $acf_excludes as $p )
716         {
717             unset( $post_types[ $p ] );
718         }
719         
720         
721         return $post_types;
722         
723     }

726     /*
727     *  get_image_sizes
728     *
729     *  @description: returns an array holding all the image sizes
730     *  @since 3.2.8
731     *  @created: 6/07/12
732     */
733     
734     function get_image_sizes( $sizes )
735     {
736         // find all sizes
737         $all_sizes = get_intermediate_image_sizes();
738         
739         
740         // define default sizes
741         $sizes = array_merge($sizes, array(
742             'thumbnail' =>  __("Thumbnail",'acf'),
743             'medium'    =>  __("Medium",'acf'),
744             'large'     =>  __("Large",'acf'),
745             'full'      =>  __("Full",'acf')
746         ));
747         
748         
749         // add extra registered sizes
750         foreach( $all_sizes as $size )
751         {
752             if( !isset($sizes[ $size ]) )
753             {
754                 $sizes[ $size ] = ucwords( str_replace('-', ' ', $size) );
755             }
756         }
757
