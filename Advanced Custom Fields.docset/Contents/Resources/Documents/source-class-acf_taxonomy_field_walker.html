 52     function load_value( $value, $post_id, $field )
 53     {
 54         if( $field['load_save_terms'] )
 55         {
 56             $value = array();
 57             
 58             $terms = get_the_terms( $post_id, $field['taxonomy'] );
 59             
 60             if( is_array($terms) ){ foreach( $terms as $term ){

 64             }}

 67

 73     /*
 74     *  update_value()
 75     *
 76     *  This filter is appied to the $value before it is updated in the db
 77     *
 78     *  @type    filter
 79     *  @since   3.6
 80     *  @date    23/01/13
 81     *
 82     *  @param   $value - the value which will be saved in the database
 83     *  @param   $field - the field array holding all the field options
 84     *  @param   $post_id - the $post_id of which the value will be saved
 85     *
 86     *  @return  $value - the modified value
 87     */
 88     
 89     function update_value( $value, $post_id, $field )
 90     {
 91         // vars
 92         if( is_array($value) )
 93         {
 94             $value = array_filter($value);
 95         }
 96         
 97         
 98         if( $field['load_save_terms'] )
 99         {
100             // Parse values
101             $value = apply_filters( 'acf/parse_types', $value );
102         
103             wp_set_object_terms( $post_id, $value, $field['taxonomy'], false );
104         }
105         
106         
107         return $value;
108     }

111     /*
112     *  format_value_for_api()
113     *
114     *  This filter is appied to the $value after it is loaded from the db and before it is passed back to the api functions such as the_field
115     *
116     *  @type    filter
117     *  @since   3.6
118     *  @date    23/01/13
119     *
120     *  @param   $value  - the value which was loaded from the database
121     *  @param   $post_id - the $post_id from which the value was loaded
122     *  @param   $field  - the field array holding all the field options
123     *
124     *  @return  $value  - the modified value
125     */
126     
127     function format_value_for_api( $value, $post_id, $field )
128     {
129         // no value?
130         if( !$value )
131         {
132             return $value;
133         }
134         
135         
136         // temp convert to array
137         $is_array = true;
138         
139         if( !is_array($value) )
140         {
141             $is_array = false;
142             $value = array( $value );
143         }
144         
145         
146         // format
147         if( $field['return_format'] == 'object' )
148         {
149             foreach( $value as $k => $v )
150             {
151                 $value[ $k ] = get_term( $v, $field['taxonomy'] );
152             }
153         }
154         
155         
156         // de-convert from array
157         if( !$is_array && isset($value[0]) )
158         {
159             $value = $value[0];
160         }
161         
162         // Note: This function can be removed if not used
163         return $value;
164     }

413     // start_el
414     function start_el( &$output, $term, $depth = 0, $args = array(), $current_object_id = 0)
415     {
416         // vars
417         $selected = in_array( $term->term_id, $this->field['value'] );
418         
419         if( $this->field['field_type'] == 'checkbox' )
420         {
421             $output .= '<li><label class="selectit"><input type="checkbox" name="' . $this->field['name'] . '" value="' . $term->term_id . '" ' . ($selected ? 'checked="checked"' : '') . ' /> ' . $term->name . '</label>';
422         }
423         elseif( $this->field['field_type'] == 'radio' )
424         {
425             $output .= '<li><label class="selectit"><input type="radio" name="' . $this->field['name'] . '" value="' . $term->term_id . '" ' . ($selected ? 'checked="checkbox"' : '') . ' /> ' . $term->name . '</label>';
426         }
427         elseif( $this->field['field_type'] == 'select' )
428         {
429             $indent = str_repeat("&mdash; ", $depth);
430             $output .= '<option value="' . $term->term_id . '" ' . ($selected ? 'selected="selected"' : '') . '>' . $indent . $term->name . '</option>';
431         }
432         
433     }
434     
435     
436     //end_el
437     function end_el( &$output, $term, $depth = 0, $args = array() )
438     {
439         if( in_array($this->field['field_type'], array('checkbox', 'radio')) )
440         {
441             $output .= '</li>';
442         }
443         
444         $output .= "\n";
445     }
