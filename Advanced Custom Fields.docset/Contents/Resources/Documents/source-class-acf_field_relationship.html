 49     /*
 50     *  load_field()
 51     *  
 52     *  This filter is appied to the $field after it is loaded from the database
 53     *  
 54     *  @type filter
 55     *  @since 3.6
 56     *  @date 23/01/13
 57     *  
 58     *  @param $field - the field array holding all the field options
 59     *  
 60     *  @return $field - the field array holding all the field options
 61     */
 62     
 63     function load_field( $field )
 64     {
 65         // validate post_type
 66         if( !$field['post_type'] || !is_array($field['post_type']) || in_array('', $field['post_type']) )
 67         {
 68             $field['post_type'] = array( 'all' );
 69         }
 70 
 71         
 72         // validate taxonomy
 73         if( !$field['taxonomy'] || !is_array($field['taxonomy']) || in_array('', $field['taxonomy']) )
 74         {
 75             $field['taxonomy'] = array( 'all' );
 76         }
 77         
 78         
 79         // validate result_elements
 80         if( !is_array( $field['result_elements'] ) )
 81         {
 82             $field['result_elements'] = array();
 83         }
 84         
 85         if( !in_array('post_title', $field['result_elements']) )
 86         {
 87             $field['result_elements'][] = 'post_title';
 88         }
 89         
 90         
 91         // filters
 92         if( !is_array( $field['filters'] ) )
 93         {
 94             $field['filters'] = array();
 95         }
 96         
 97         
 98         // return
 99         return $field;
100     }

103     /*
104     *  posts_where
105     *
106     *  @description: 
107     *  @created: 3/09/12
108     */
109     
110     function posts_where( $where, &$wp_query )
111     {
112         global $wpdb;
113         
114         if ( $title = $wp_query->get('like_title') )
115         {
116             $where .= " AND " . $wpdb->posts . ".post_title LIKE '%" . esc_sql( like_escape(  $title ) ) . "%'";
117         }
118         
119         return $where;
120     }

123     /*
124     *  query_posts
125     *
126     *  @description: 
127     *  @since: 3.6
128     *  @created: 27/01/13
129     */
130     
131     function query_posts()
132     {
133         // vars
134         $r = array(
135             'next_page_exists' => 1,
136             'html' => ''
137         );
138         
139         
140         // options
141         $options = array(
142             'post_type'                 =>  'all',
143             'taxonomy'                  =>  'all',
144             'posts_per_page'            =>  10,
145             'paged'                     =>  1,
146             'orderby'                   =>  'title',
147             'order'                     =>  'ASC',
148             'post_status'               =>  'any',
149             'suppress_filters'          =>  false,
150             's'                         =>  '',
151             'lang'                      =>  false,
152             'update_post_meta_cache'    =>  false,
153             'field_key'                 =>  '',
154             'nonce'                     =>  '',
155             'ancestor'                  =>  false,
156         );
157         
158         $options = array_merge( $options, $_POST );
159         
160         
161         // validate
162         if( ! wp_verify_nonce($options['nonce'], 'acf_nonce') )
163         {
164             die();
165         }
166         
167         
168         // WPML
169         if( $options['lang'] )
170         {
171             global $sitepress;
172             
173             if( !empty($sitepress) )
174             {
175                 $sitepress->switch_lang( $options['lang'] );
176             }
177         }
178         
179         
180         // convert types
181         $options['post_type'] = explode(',', $options['post_type']);
182         $options['taxonomy'] = explode(',', $options['taxonomy']);
183         
184         
185         // load all post types by default
186         if( in_array('all', $options['post_type']) )
187         {
188             $options['post_type'] = apply_filters('acf/get_post_types', array());
189         }
190         
191         
192         // attachment doesn't work if it is the only item in an array???
193         if( is_array($options['post_type']) && count($options['post_type']) == 1 )
194         {
195             $options['post_type'] = $options['post_type'][0];
196         }
197         
198         
199         // create tax queries
200         if( ! in_array('all', $options['taxonomy']) )
201         {
202             // vars
203             $taxonomies = array();
204             $options['tax_query'] = array();
205             
206             foreach( $options['taxonomy'] as $v )
207             {
208                 
209                 // find term (find taxonomy!)
210                 // $term = array( 0 => $taxonomy, 1 => $term_id )
211                 $term = explode(':', $v); 
212                 
213                 
214                 // validate
215                 if( !is_array($term) || !isset($term[1]) )
216                 {
217                     continue;
218                 }
219                 
220                 
221                 // add to tax array
222                 $taxonomies[ $term[0] ][] = $term[1];
223                 
224             }
225             
226             
227             // now create the tax queries
228             foreach( $taxonomies as $k => $v )
229             {
230                 $options['tax_query'][] = array(
231                     'taxonomy' => $k,
232                     'field' => 'id',
233                     'terms' => $v,
234                 );
235             }
236         }
237         
238         unset( $options['taxonomy'] );
239         
240         
241         // search
242         if( $options['s'] )
243         {
244             $options['like_title'] = $options['s'];
245             
246             add_filter( 'posts_where', array($this, 'posts_where'), 10, 2 );
247         }
248         
249         unset( $options['s'] );
250         
251         
252         // load field
253         $field = array();
254         if( $options['ancestor'] )
255         {
256             $ancestor = apply_filters('acf/load_field', array(), $options['ancestor'] );
257             $field = acf_get_child_field_from_parent_field( $options['field_key'], $ancestor );
258         }
259         else
260         {
261             $field = apply_filters('acf/load_field', array(), $options['field_key'] );
262         }
263         
264         
265         // get the post from which this field is rendered on
266         $the_post = get_post( $options['post_id'] );
267         
268         
269         // filters
270         $options = apply_filters('acf/fields/relationship/query', $options, $field, $the_post);
271         $options = apply_filters('acf/fields/relationship/query/name=' . $field['_name'], $options, $field, $the_post );
272         $options = apply_filters('acf/fields/relationship/query/key=' . $field['key'], $options, $field, $the_post );
273         
274         
275         // query
276         $wp_query = new WP_Query( $options );
277 
278         
279         // global
280         global $post;
281         
282         
283         // loop
284         while( $wp_query->have_posts() )
285         {
286             $wp_query->the_post();
287             
288             
289             // right aligned info
290             $title = '<span class="relationship-item-info">';
291                 
292                 if( in_array('post_type', $field['result_elements']) )
293                 {
294                     $post_type_object = get_post_type_object( get_post_type() );
295                     $title .= $post_type_object->labels->singular_name;
296                 }
297                 
298                 // WPML
299                 if( $options['lang'] )
300                 {
301                     $title .= ' (' . $options['lang'] . ')';
302                 }
303                 
304             $title .= '</span>';
305             
306             
307             // featured_image
308             if( in_array('featured_image', $field['result_elements']) )
309             {
310                 $image = get_the_post_thumbnail( get_the_ID(), array(21, 21) );
311                 
312                 $title .= '<div class="result-thumbnail">' . $image . '</div>';
313             }
314             
315             
316             // title
317             $title .= get_the_title();
318             
319             
320             // status
321             if( get_post_status() != "publish" )
322             {
323                 $title .= ' (' . get_post_status() . ')';
324             }
325                 
326             
327             // filters
328             $title = apply_filters('acf/fields/relationship/result', $title, $post, $field, $the_post);
329             $title = apply_filters('acf/fields/relationship/result/name=' . $field['_name'] , $title, $post, $field, $the_post);
330             $title = apply_filters('acf/fields/relationship/result/key=' . $field['key'], $title, $post, $field, $the_post);
331             
332             
333             // update html
334             $r['html'] .= '<li><a href="' . get_permalink() . '" data-post_id="' . get_the_ID() . '">' . $title .  '<span class="acf-button-add"></span></a></li>';
335         }
336         
337         
338         if( (int)$options['paged'] >= $wp_query->max_num_pages )
339         {
340             $r['next_page_exists'] = 0;
341         }
342         
343         
344         wp_reset_postdata();
345         
346         
347         // return JSON
348         echo json_encode( $r );
349         
350         die();
351             
352     }

702     /*
703     *  format_value()
704     *
705     *  This filter is appied to the $value after it is loaded from the db and before it is passed to the create_field action
706     *
707     *  @type    filter
708     *  @since   3.6
709     *  @date    23/01/13
710     *
711     *  @param   $value  - the value which was loaded from the database
712     *  @param   $post_id - the $post_id from which the value was loaded
713     *  @param   $field  - the field array holding all the field options
714     *
715     *  @return  $value  - the modified value
716     */
717     
718     function format_value( $value, $post_id, $field )
719     {
720         // empty?
721         if( !empty($value) )
722         {
723             // Pre 3.3.3, the value is a string coma seperated
724             if( is_string($value) )
725             {
726                 $value = explode(',', $value);
727             }
728             
729             
730             // convert to integers
731             if( is_array($value) )
732             {
733                 $value = array_map('intval', $value);
734                 
735                 // convert into post objects
736                 $value = $this->get_posts( $value );
737             }
738             
739         }
740         
741         
742         // return value
743         return $value;  
744     }

747     /*
748     *  format_value_for_api()
749     *
750     *  This filter is appied to the $value after it is loaded from the db and before it is passed back to the api functions such as the_field
751     *
752     *  @type    filter
753     *  @since   3.6
754     *  @date    23/01/13
755     *
756     *  @param   $value  - the value which was loaded from the database
757     *  @param   $post_id - the $post_id from which the value was loaded
758     *  @param   $field  - the field array holding all the field options
759     *
760     *  @return  $value  - the modified value
761     */
762     
763     function format_value_for_api( $value, $post_id, $field )
764     {
765         // empty?
766         if( !$value )
767         {
768             return $value;
769         }
770         
771

805     *
806     *  This function will take an array of post_id's ($value) and return an array of post_objects
807     *
808     *  @type    function
809     *  @date    7/08/13

811     *  @param   $post_ids (array) the array of post ID's
812     *  @return  (array) an array of post objects
813     */
814     
815     function get_posts( $post_ids )
816     {
817         // validate
818         if( empty($post_ids) )
819         {
820             return $post_ids;
821         }
822         
823         
824         // vars
825         $r = array();
826         
827         
828         // find posts (DISTINCT POSTS)
829         $posts = get_posts(array(
830             'numberposts'   =>  -1,
831             'post__in'      =>  $post_ids,
832             'post_type'     =>  apply_filters('acf/get_post_types', array()),
833             'post_status'   =>  'any',
834         ));
835 
836         
837         $ordered_posts = array();
838         foreach( $posts as $p )
839         {
840             // create array to hold value data
841             $ordered_posts[ $p->ID ] = $p;
842         }
843         
844         
845         // override value array with attachments
846         foreach( $post_ids as $k => $v)
847         {
848             // check that post exists (my have been trashed)
849             if( isset($ordered_posts[ $v ]) )
850             {
851                 $r[] = $ordered_posts[ $v ];
852             }
853         }
854         
855         
856         // return
857         return $r;
